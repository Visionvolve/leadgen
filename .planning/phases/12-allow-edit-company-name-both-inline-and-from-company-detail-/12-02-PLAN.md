      """Company-name normalization for duplicate detection."""
      from __future__ import annotations
      import re
      import unicodedata

      # Canonical lowercased forms. Ordered longest-first because we strip
      # the LAST matching trailing token until none match — 'spol s r o'
      # must be tried before 's r o'.
      LEGAL_SUFFIXES: list[str] = [
          "spol s r o", "spol s ro",
          "s r o", "sro",
          "v o s", "vos",
          "a s", "as",
          "k s", "ks",
          "gmbh", "mbh",
          "limited", "ltd",
          "company", "corp", "inc",
          "llc", "ag", "kg", "co",
      ]

      # Punctuation to strip (replaced by space): , . & ( ) [ ] ' " / \\
      _PUNCT_TO_SPACE_RE = re.compile(r"[,.&()\\[\\]'\"/\\\\\\-_]")
      _WS_RE = re.compile(r"\\s+")

      def normalize_company_name(s: str | None) -> str:
          if s is None:
              return ""
          # 1. trim + collapse whitespace
          s = s.strip()
          if not s:
              return ""
          # 2. lowercase
          s = s.lower()
          # 3. NFKD + strip combining marks
          s = unicodedata.normalize("NFKD", s)
          s = "".join(ch for ch in s if unicodedata.category(ch) != "Mn")
          # 4. replace punctuation with space (hyphen + underscore included)
          s = _PUNCT_TO_SPACE_RE.sub(" ", s)
          # 5. collapse whitespace
          s = _WS_RE.sub(" ", s).strip()
          # 6. strip trailing legal suffixes iteratively
          while True:
              stripped = False
              for suf in LEGAL_SUFFIXES:
                  # match suffix as the trailing TOKEN(s) only
                  if s == suf:
                      s = ""
                      stripped = True
                      break
                  if s.endswith(" " + suf):
                      s = s[: -(len(suf) + 1)].rstrip()
                      stripped = True
                      break
              if not stripped:
                  break
          # 7. final collapse + trim
          s = _WS_RE.sub(" ", s).strip()
          return s

      #!/usr/bin/env python3
      """Backfill companies.normalized_name for BL-1117.

      Idempotent: only updates rows whose stored normalized_name differs
      from normalize_company_name(name). Batched by id ranges to keep
      individual transactions short. Safe to re-run after a partial
      failure.
      """
      from __future__ import annotations

      import os
      import sys
      from pathlib import Path

      # Make the api package importable when running standalone
      sys.path.insert(0, str(Path(__file__).resolve().parent.parent))

      from sqlalchemy import create_engine, text
      from api.services.name_normalize import normalize_company_name

      BATCH = 500

      def main() -> int:
          url = os.environ.get("LEADGEN_DATABASE_URL")
          if not url:
              print("LEADGEN_DATABASE_URL not set", file=sys.stderr)
              return 1
          eng = create_engine(url, pool_pre_ping=True)
          updated = 0
          seen = 0
          with eng.begin() as conn:
              rows = conn.execute(text(
                  "SELECT id, name, normalized_name FROM companies ORDER BY id"
              )).fetchall()
              seen = len(rows)
          # batch updates outside the read loop
          for i in range(0, len(rows), BATCH):
              chunk = rows[i:i+BATCH]
              changes = []
              for r in chunk:
                  desired = normalize_company_name(r[1])
                  if (r[2] or "") != desired:
                      changes.append((r[0], desired))
              if not changes:
                  continue
              with eng.begin() as conn:
                  for cid, nn in changes:
                      conn.execute(text(
                          "UPDATE companies SET normalized_name = :nn WHERE id = :id"
                      ), {"nn": nn, "id": cid})
              updated += len(changes)
              print(f"  ... batch {i//BATCH + 1}: {len(changes)} updated")
          print(f"Done. seen={seen} updated={updated}")
          return 0

      if __name__ == "__main__":
          raise SystemExit(main())

      from sqlalchemy import func
      from ..models import Company, Contact, Owner, db

      def find_name_collisions(
          tenant_id: str,
          normalized_name: str,
          exclude_id: str | None = None,
      ) -> list[dict]:
          """Tenant-scoped exact-match lookup on normalized_name.

          Returns rich summaries for the UI to render in
          DuplicateCompanyModal. Empty normalized_name short-circuits to
          [] without touching the DB.
          """
          if not normalized_name:
              return []
          q = (
              db.session.query(
                  Company.id,
                  Company.name,
                  Company.domain,
                  Company.status,
                  Company.owner_id,
                  Owner.name.label("owner_name"),
                  func.count(Contact.id).label("contact_count"),
                  func.max(Contact.updated_at).label("last_activity_at"),
              )
              .outerjoin(Owner, Owner.id == Company.owner_id)
              .outerjoin(Contact, Contact.company_id == Company.id)
              .filter(Company.tenant_id == str(tenant_id))
              .filter(Company.normalized_name == normalized_name)
              .group_by(Company.id, Owner.name)
          )
          if exclude_id:
              q = q.filter(Company.id != str(exclude_id))
          out = []
          for row in q.all():
              out.append({
                  "id": row[0],
                  "name": row[1],
                  "domain": row[2],
                  "status": row[3],
                  "owner": {"id": row[4], "name": row[5]} if row[4] else None,
                  "contact_count": int(row[6] or 0),
                  "last_activity_at": row[7].isoformat() if row[7] else None,
              })
          return out
