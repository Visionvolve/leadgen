    Create frontend/src/hooks/useCompanyDuplicateGate.ts:

      import { useState, useEffect, useCallback } from 'react'

      export type CompanyMatch = {
        id: string
        name: string
        domain: string | null
        status: string | null
        owner: { id: string; name: string } | null
        contact_count: number
        last_activity_at: string | null
      }

      export type DuplicateContext = {
        editedCompanyId: string
        attemptedName: string
        matches: CompanyMatch[]
        resolveMerge: (intoId: string) => Promise<void>
        resolveUseExisting: (matchId: string) => void
        resolveKeepBoth: () => Promise<void>
        resolveCancel: () => void
      }

      type EventDetail = {
        editedCompanyId: string
        attemptedName: string
        matches: CompanyMatch[]
        retryWithKeepBoth: () => Promise<void>
        revertInput: () => void
        afterMerge: () => void
      }

      /**
       * Window-event subscriber. useInlineEdit dispatches
       * 'leadgen:company-duplicate' on 409; this hook holds the resulting
       * state and exposes resolution callbacks for DuplicateCompanyModal.
       */
      export function useCompanyDuplicateGate() {
        const [pending, setPending] = useState<DuplicateContext | null>(null)

        useEffect(() => {
          const handler = (e: Event) => {
            const detail = (e as CustomEvent<EventDetail>).detail
            const ctx: DuplicateContext = {
              editedCompanyId: detail.editedCompanyId,
              attemptedName: detail.attemptedName,
              matches: detail.matches,
              resolveMerge: async (intoId: string) => {
                const res = await fetch(
                  `/api/companies/${detail.editedCompanyId}/merge?into=${encodeURIComponent(intoId)}`,
                  { method: 'POST', credentials: 'include' },
                )
                if (!res.ok) {
                  const body = await res.json().catch(() => ({}))
                  throw new Error(body.error || 'merge failed')
                }
                detail.afterMerge()
                setPending(null)
                // navigate to surviving record
                window.dispatchEvent(new CustomEvent('leadgen:navigate', {
                  detail: `/companies/${intoId}`,
                }))
              },
              resolveUseExisting: (matchId: string) => {
                detail.revertInput()
                setPending(null)
                window.dispatchEvent(new CustomEvent('leadgen:navigate', {
                  detail: `/companies/${matchId}`,
                }))
              },
              resolveKeepBoth: async () => {
                await detail.retryWithKeepBoth()
                setPending(null)
              },
              resolveCancel: () => {
                detail.revertInput()
                setPending(null)
              },
            }
            setPending(ctx)
          }
          window.addEventListener('leadgen:company-duplicate', handler)
          return () => window.removeEventListener('leadgen:company-duplicate', handler)
        }, [])

        return { pendingDuplicate: pending, dismiss: () => setPending(null) }
      }

    Run `cd frontend && npx tsc --noEmit`.

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| browser → API | Inline-edit PATCH; merge POST; both go through credentialed fetch |
| event bus | window CustomEvent — same-origin only, no cross-origin attack surface |
| confirm_duplicate query | Client-controlled, server already validates the literal 'keep_both' string |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-12-26 | S (Event spoofing) | leadgen:company-duplicate CustomEvent | accept | Same-origin script execution would mean attacker already has code-exec; CSP defends. Listener does not auth — it trusts in-page events. |
| T-12-27 | T (Tampering with confirm_duplicate query) | client → server | mitigate | Server treats the param as a literal string 'keep_both' and writes an audit row regardless; no privilege effect. |
| T-12-28 | I (Match list leaked to client) | 409 response | accept | Matches are scoped to caller's tenant by the server; same trust level as Companies list. |
| T-12-29 | D (UI lockup on errors) | merge fetch | mitigate | resolveMerge awaits the fetch; thrown errors propagate up so the modal can show an error toast (Plan 12-07). |
</threat_model>
