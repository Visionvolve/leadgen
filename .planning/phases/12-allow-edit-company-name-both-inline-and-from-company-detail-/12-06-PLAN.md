        label?: string
        name: string
        value: string
        onSave: (newValue: string) => Promise<void>  // async; may throw
        onCancel?: () => void
        className?: string
        placeholder?: string
        variant?: 'heading' | 'inline'   // 'heading' uses text-2xl font-bold

        Show input, Save button, Cancel button, optional error span.
        Disable buttons while saving === true.
        Save button label flips to 'Saving...' while in-flight.

    Use plain React text node interpolation for the value -- React escapes

               await mutation.mutateAsync({ id: company.id, data: { name: newValue } })

               // 409: hand off to the shared gate hook by dispatching
               // the same custom event used by useInlineEdit.
               if (err?.status === 409 && err?.code === 'duplicate_company_name') {
                 const matches = err.details?.matches ?? []
                 return new Promise<void>((resolve, reject) => {
                   window.dispatchEvent(new CustomEvent('leadgen:company-duplicate', {
                     detail: {
                       editedCompanyId: company.id,
                       attemptedName: newValue,
                       matches,
                       retryWithKeepBoth: async () => {
                         await mutation.mutateAsync({
                           id: company.id,
                           data: { name: newValue },
                           params: { confirm_duplicate: 'keep_both' },
                         })
                         resolve()
                       },
                       revertInput: () => reject(new Error('Cancelled')),
                       afterMerge: () => resolve(),
                     },
                   }))
                 })
               }
               throw err
             }
